Algorithm :-
Step 1:-	Begin:
Step 2:- 		Read   x1,x2,x3,t1,t2,fx
			Read    a,b,p1,p2,c 
Step 3:-		do
  			t1=x1
			t2=x2
Step 4:-		comput  fx3=(a*(pow(x3,p1)))+(b*(pow(x3,p2)))+(c)
Step 5:-		if(fx3<0) then
			x1=x3
			else
			x2=x3
Step 6:- 		while((t1!=x1) || (t2!=x2))
  			write x1,x2
Step 7:-	 End
